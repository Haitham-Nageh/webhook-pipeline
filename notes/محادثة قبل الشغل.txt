نشرح كل جزء ببساطة
src/app.ts

فيه إعدادات express:

middlewares

routes

error handler

src/server.ts

هذا الملف يشغل الـ API server

يعني:

يعمل listen على port

src/worker.ts

هذا الملف يشغل الـ worker

يعني:

loop أو polling

يجيب jobs pending

ينفذها

config/

للأشياء العامة مثل:

environment variables

constants

retry limits

timeouts

routes/

تعريف المسارات فقط

مثلاً:

/pipelines

/webhooks/:id

/jobs/:id

controllers/

تستقبل الطلب والرد

يعني controller لا يحتوي business logic كثير
فقط:

يأخذ request

يستدعي service

يرجع response

services/

هذا أهم جزء
هنا المنطق الحقيقي للمشروع

مثلاً:

pipeline.service.ts

إنشاء pipeline

تعديلها

حذفها

webhook.service.ts

استقبال webhook

إنشاء job

job.service.ts

قراءة حالة job

عرض history

processing.service.ts

اختيار الـ action المناسبة وتنفيذها

delivery.service.ts

إرسال النتيجة للـ subscribers

retry logic

تسجيل attempts

actions/

هنا أنواع المعالجة

مثلاً:

add-timestamp.action.ts

يضيف وقت على payload

filter-fields.action.ts

يبقي فقط fields معينة

transform-payload.action.ts

يغير شكل البيانات

هذا التقسيم ممتاز لأنه يجعل إضافة action جديدة سهلة.

repositories/

هنا التعامل مع قاعدة البيانات

مثلاً:

create pipeline

create job

update job status

create delivery attempt

ليش مفيد؟
لأنه يمنع خلط DB queries مع business logic.

lib/

للأدوات الجاهزة المشتركة

مثل:

Prisma client

Axios instance

Logger

middleware/

أشياء تعمل بين request وresponse

مثل:

error handler

validate request body

schemas/

هنا Zod schemas

مثلاً:

schema لإنشاء pipeline

schema لتحديث pipeline

types/

إذا عندك أنواع مخصصة
أو enums
أو interfaces إضافية

utils/

دوال مساعدة صغيرة

مثل:

sleep

retry helpers

parsing helpers

prisma/

خاص بقاعدة البيانات

schema.prisma

تعريف الجداول والعلاقات

tests/

للاختبارات

حتى لو تعمل 3 أو 4 اختبارات فقط، هذا يعطي انطباع ممتاز.

docker/

ملفات docker لكل service

مثلاً:

Dockerfile للـ API

Dockerfile للـ Worker
